#include <avr/io.h> 
#include <avr/sleep.h>
int val = 0;

void setup()
{
  TCCR1A = 0; 
  TCCR1B = 0; 
  OCR1A = 15624; 
  TCCR1B |= (1 << 0);
  TCCR1B |= (1 << 2);
  TCCR1B |= (1 << 3);
  TCCR1B |= ((1<<0)|(1<<2)| (1<<3)); 
  DDRB |= (1 << 5);
  DDRB |= (0 << 1);
}

void loop()
{
  if (PINB == (1 << 0))
  {
    val++;
    if (val > 2) { val = 0; }
  }

  if (val == 0)
  {
  PORTB |= (1 << 5);

  PORTB &= ~(1 << 5);

  }
  if (val == 1)
  {
  PORTB |= (1 << 5);
  
  PORTB &= ~(1 << 5);
  
  }
  if (val == 2)
  {
  PORTB |= (1 << 5);
  
  PORTB &= ~(1 << 5);
  
  }
}
  ISR(TIMER1_OVF_vect)  
{
  PORTB = (~PINB&(1<<0));
}
ISR(TIMER1_COMPA_vect) 
{ 
  PORTB = (~PINB&(1<<5));
}

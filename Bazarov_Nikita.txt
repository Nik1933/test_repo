#include <avr/io.h>
#include <avr/interrupt.h>
#define LEDPIN 13

void setup()
{
    DDRB &= ~(0 << 5)
    cli();
    TCCR1A = 0;
    TCCR1B = 0;
    OCR1A = 15624;
    TCCR1B |= (1 << WGM12);
    TCCR1B |= (1 << CS10);
    TCCR1B |= (1 << CS12);
    TIMSK1 |= (1 << OCIE1A);
    sei();
  	PORTB = 0;
}

void loop()
{
    
}

ISR(TIMER1_COMPA_vect)
{
    PORTB |= (1 << 5), PORTB &= ~(1 << 5);
}
